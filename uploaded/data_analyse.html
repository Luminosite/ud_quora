def show_points(results, value_column='y',
                title="mean target label's distribution on ln(len_q1/len_q2)",
                x_l="ln(len_q1/len_q2)",
                y_l = "mean target"):
    x = results.index*0.04
    y = results[value_column]
    pl.figure(figsize=(15,5))
    pl.scatter(x, y, marker = 'o', color ='blue', s=5, label="distribution")
    pl.legend(loc = 'best')
    pl.suptitle(title, fontsize=16)
    pl.xlabel(x_l)
    pl.ylabel(y_l)
    pl.show()

def hist(d, b=40, xl="consine value", yl="count", t="consine histogram"):
    pl.figure(figsize=(15,5))
    pl.hist(d, bins=b)
    pl.xlabel(xl)

def draw(dist):
    cd = tData[dist].apply(lambda x: int(x/0.01))
    cd = pd.DataFrame({'x': cd * 0.01, 'y': tData['is_duplicate']})
    gd = cd.groupby("x").mean()
    draw_bars(gd, title=("target's mean value distribution on %s" % dist), x_label=dist)
    hist(tData[dist])

for dist in  other_dists:
    draw(dist)
